(
id INT AUTO_INCREMENT,
item_name VARCHAR(70) NOT NULL UNIQUE,
item_price INT NOT NULL,
item_description VARCHAR(255) NOT NULL ,
item_image VARCHAR(255) NOT NULL,
PRIMARY KEY(id)
);
